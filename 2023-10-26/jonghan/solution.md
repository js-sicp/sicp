# 연습 문제
## 1.
```js
console.log(10); // 10
console.log(5 + 3 + 4); // 12
console.log(9 - 1); // 8
console.log(6 / 2); // 3
console.log(2 * 4 + (4 - 6)); // 6

const a = 3;
const b = a + 1;
console.log(a + b + a * b); // 19
console.log(a === b); // false
console.log(b > a && b < a * b ? b : a); // 4
const answer = a === 4 ? 6 : b === 4 ? 6 + 7 + a : 25;
console.log(answer); // 16
```
